245void set_addressing(uint8_t pagenum, uint8_t c_start) {
246 uint8_t a, b, c;
247 c = c_start + oled_shift;
248 Soft_I2C_Start();
249 Soft_I2C_Write(led_addr); // device addres
250 Soft_I2C_Write(0); // 0 - continious mode, command; 64 - Co, data
251 Soft_I2C_Write(0xB0 + pagenum); // set page number
252
253 if (c <= 15) {
254 a = c;
255 b = 0;
256 } else {
257 b = c / 16;
258 a = c - b * 16;
259 }
260 Soft_I2C_Write(a); // set lower nibble of start address
261 Soft_I2C_Write(0x10 + b); // set higher nibble of start address
262
263 Soft_I2C_Start();
264 Soft_I2C_Write(led_addr); // device addres
265 Soft_I2C_Write(64); // 0 - continious mode, command; 64 - Co, data
266}
