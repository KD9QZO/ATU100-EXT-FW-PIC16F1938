117adc_result_t ADC_GetConversion(adc_channel_t channel)
118{
119 // select the A/D channel
120 ADCON0bits.CHS = channel;
121
122 // Turn on the ADC module
123 ADCON0bits.ADON = 1;
124
125 // Acquisition time delay
126 __delay_us(ACQ_US_DELAY);
127
128 // Start the conversion
129 ADCON0bits.GO_nDONE = 1;
130
131 // Wait for the conversion to finish
132 while (ADCON0bits.GO_nDONE)
133 {
134 }
135
136 // Conversion finished, return the result
137 return ((adc_result_t)((ADRESH << 8) + ADRESL));
138}
